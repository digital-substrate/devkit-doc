# DevKit Documentation — build dependencies
#
# Install with: pip install -r requirements.txt

# Sphinx documentation generator
sphinx

# Furo theme (clean, responsive)
furo

# MyST parser (Markdown support in Sphinx)
myst-parser

# Copy-button on every code block
sphinx-copybutton

# Markdown builder (used by `make notebooklm`)
sphinx-markdown-builder

# Sitemap generator — emits build/html/sitemap.xml for crawlers
sphinx-sitemap

# OpenGraph + meta-description tags in <head> for SEO and link previews
sphinxext-opengraph

# dsviper runtime (required by pyi_signatures extension to extract API type hints)
dsviper

# sphinx-js — generates the Node API reference (dsviper-node/api/) from the
# @digitalsubstrate/dsviper TypeScript declarations via TypeDoc. Needs Node +
# the root `npm install` (see CLAUDE.md Build Commands).
sphinx-js

# numpy — imported by the blobs.md doctests (`make check`); without it the
# numpy-interop examples fail on a clean environment
numpy
